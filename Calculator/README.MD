#include <Keypad.h>

const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] = {
  { '1', '2', '3', 'A' },
  { '7', '8', '9', 'B' },
  { '4', '5', '6', 'C' },
  { '*', '0', '#', 'D' }

Keypad kpd = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

  lcd.begin(16, 2);
  lcd.print("DIY Calculator");
  lcd.setCursor(0, 1);

void loop() {
  key = kpd.getKey();

  if (key != NO_KEY)
    DetectButtons();

  if (result == true)
    CalculateResult();

  DisplayResult();
}
